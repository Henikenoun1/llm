// tunisianPatterns.ts
//
// Hybrid intent router for Tounsi / FR / Arabizi / Arabic small-talk.
//
// Pipeline (call order from your app):
//   1. normalizeTunisian(input)          -> texte propre
//   2. detectIntentDetailed(input)       -> { intent, confidence, source }
//        - "regex"        : match franc, on peut répondre direct
//        - "regex_low"    : match mais texte contient des mots métier
//                           => on RECOMMANDE de laisser le LLM décider
//        - "fallback"     : aucune regex, intent = "unknown"
//   3. Si source === "regex"  -> replyTunisian(input)
//      Si source === "regex_low" ou intent === "unknown"
//                                -> passer au LLM intent classifier
//
// L'API publique historique est conservée à l'identique :
//   - export type Intent
//   - export function normalizeTunisian(text)
//   - export function detectIntent(input): Intent
//   - export function replyTunisian(input): string
// => l'intégration existante continue de marcher sans modification.
//
// Nouveautés (sans casser l'existant) :
//   - export function detectIntentDetailed(input): IntentResult
//   - export function shouldDeferToLLM(input): boolean
//   - normalisation arabe enrichie (أإآ→ا, ى→ي, ة→ه, tashkeel, tatweel…)
//   - garde-fou anti faux-positifs : "ça va ma commande ?" n'est PAS how_are_you
//   - priorité claire : bot_identity > thanks > goodbye > how_are_you > greeting
//   - exemples ajoutés (yo, hey, hola, à plus, see you, ahla bik, sbah el khir, …)

export type Intent =
  | "greeting"
  | "bot_identity"
  | "how_are_you"
  | "thanks"
  | "goodbye"
  | "unknown";

export type IntentSource = "regex" | "regex_low" | "fallback";

export interface IntentResult {
  intent: Intent;
  confidence: number; // 0..1
  source: IntentSource;
  matched?: string;   // mot/expression qui a déclenché le match (debug)
}

// ─────────────────────────────────────────────────────────────────────────────
// 1) NORMALISATION
// ─────────────────────────────────────────────────────────────────────────────

export function normalizeTunisian(text: string): string {
  if (!text) return "";

  let t = text;

  // 1. Unicode NFD + suppression des diacritiques latins (é → e)
  t = t.normalize("NFD").replace(/[\u0300-\u036f]/g, "");

  // 2. Tashkeel arabe (fatha, damma, kasra, sukun, shadda, tanwin)
  t = t.replace(/[\u064B-\u065F\u0670]/g, "");

  // 3. Tatweel ـ
  t = t.replace(/\u0640/g, "");

  // 4. Normalisation alphabétique arabe
  t = t
    .replace(/[\u0623\u0625\u0622\u0671]/g, "\u0627") // أ إ آ ٱ → ا
    .replace(/\u0649/g, "\u064A")                       // ى → ي
    .replace(/\u0629/g, "\u0647")                       // ة → ه
    .replace(/\u0624/g, "\u0648")                       // ؤ → و
    .replace(/\u0626/g, "\u064A");                      // ئ → ي

  // 5. Lowercase + ponctuation → espace
  t = t.toLowerCase();
  t = t.replace(/[؟?!.,،؛:;'"`(){}\[\]<>«»…—–]/g, " ");

  // 6. Réduire les répétitions extrêmes (ahlaaaa → ahlaa, hhhhhh → hh)
  t = t.replace(/([a-z\u0600-\u06FF])\1{2,}/g, "$1$1");

  // 7. Espaces multiples
  t = t.replace(/\s+/g, " ").trim();

  return t;
}

// ─────────────────────────────────────────────────────────────────────────────
// 3) COMPILATION DES REGEX (avec frontières mots Unicode-safe)
// ─────────────────────────────────────────────────────────────────────────────

function escapeRegex(s: string): string {
  return s.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
}

// Les bornes \b ne marchent pas sur l'arabe → on encadre par
// "début de chaîne / espace" et "fin de chaîne / espace / ponctuation déjà
// transformée en espace lors de la normalisation".
function makeRegex(words: string[]): RegExp {
  const norm = Array.from(
    new Set(words.map((w) => normalizeTunisian(w)).filter(Boolean))
  );
  // tri par longueur décroissante : "ahla bik" doit matcher avant "ahla"
  norm.sort((a, b) => b.length - a.length);
  const escaped = norm.map(escapeRegex);
  return new RegExp(`(?:^|\\s)(${escaped.join("|")})(?:\\s|$)`, "iu");
}

// PRIORITÉ : bot_identity > thanks > goodbye > how_are_you > greeting
const patterns: { intent: Intent; regex: RegExp }[] = [
  { intent: "bot_identity", regex: makeRegex(identityWords) },
  { intent: "thanks",       regex: makeRegex(thanksWords) },
  { intent: "goodbye",      regex: makeRegex(goodbyeWords) },
  { intent: "how_are_you",  regex: makeRegex(howAreYouWords) },
  { intent: "greeting",     regex: makeRegex(greetingWords) },
];

const businessRegex = makeRegex(businessKeywords);

// Bruit / messages quasi vides → unknown direct
const noiseRegex = /^(?:wa|w|a{1,5}|h{2,}|ه{2,}|ok|oki|kk|k|\?|؟)$/iu;

// Intents "soft" qui peuvent être démotés s'il y a un mot métier à côté
const SOFT_INTENTS: Set<Intent> = new Set([
  "greeting",
  "how_are_you",
  "thanks",
]);

// ─────────────────────────────────────────────────────────────────────────────
// 4) API PUBLIQUE
// ─────────────────────────────────────────────────────────────────────────────

export function detectIntentDetailed(input: string): IntentResult {
  const text = normalizeTunisian(input);

  if (!text || noiseRegex.test(text)) {
    return { intent: "unknown", confidence: 0.2, source: "fallback" };
  }

  for (const p of patterns) {
    const m = p.regex.exec(text);
    if (!m) continue;

    const matched = m[1];

    // Anti faux-positifs : "ça va ma commande ?", "merci pour la livraison",
    // "salut, où est ma commande" → on laisse le LLM décider.
    if (SOFT_INTENTS.has(p.intent) && businessRegex.test(text)) {
      return {
        intent: p.intent,
        confidence: 0.45,
        source: "regex_low",
        matched,
      };
    }

    return {
      intent: p.intent,
      confidence: 0.92,
      source: "regex",
      matched,
    };
  }

  return { intent: "unknown", confidence: 0.1, source: "fallback" };
}

// API historique : conservée à l'identique pour ne rien casser.
export function detectIntent(input: string): Intent {
  return detectIntentDetailed(input).intent;
}

// Helper pour le routeur : true => ne pas répondre via replyTunisian,
// laisser le LLM (intent classifier + génération) prendre la main.
export function shouldDeferToLLM(input: string): boolean {
  const r = detectIntentDetailed(input);
  return r.source !== "regex" || r.intent === "unknown";
}

export function detectIntent(input: string): Intent {

  return "unknown";
}
